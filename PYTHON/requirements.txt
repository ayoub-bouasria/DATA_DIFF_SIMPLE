# DATA DIFF PYTHON - Dependencies
# ================================
# Compatible with Python 3.9-3.13
# Using datacompy with Snowpark for remote comparison

# Core comparison library
datacompy>=0.14.0

# Snowpark for remote Snowflake comparison (Python 3.9-3.13)
# This is the recommended way to compare large tables
snowflake-snowpark-python>=1.35.0

# Data manipulation
pandas>=2.2.0
numpy>=2.1.0

# Configuration management
python-dotenv>=1.0.0

# Excel export support
openpyxl>=3.1.0
xlsxwriter>=3.2.0

# CLI
click>=8.1.0

# YAML config support
pyyaml>=6.0.1

# ============================================================
# INSTALLATION NOTES
# ============================================================
#
# For Python 3.9-3.13 (recommended):
#   pip install -r requirements.txt
#
# For Python 3.14+ (Snowpark not available):
#   # Only local file comparison will work
#   pip install datacompy pandas click pyyaml python-dotenv openpyxl xlsxwriter
#
# Using conda (alternative):
#   conda create -n datadiff python=3.11
#   conda activate datadiff
#   conda install -c conda-forge snowflake-snowpark-python
#   pip install datacompy pandas click pyyaml python-dotenv openpyxl xlsxwriter
#
# ============================================================
# USAGE MODES
# ============================================================
#
# 1. Snowpark mode (remote comparison on Snowflake):
#    - Requires Python 3.9-3.13
#    - Requires snowflake-snowpark-python
#    - Best for large tables (no data download)
#    - Usage: python run_comparison.py compare TABLE1 TABLE2 --pk ID --mode snowpark
#
# 2. Local mode (compare CSV/Excel files):
#    - Works on any Python version
#    - No Snowflake connection required
#    - Usage: python run_comparison.py compare file1.csv file2.csv --pk ID
#
